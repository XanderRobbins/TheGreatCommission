# Load everything
from api_client import ScriptureTranslationClient
from models.base import ScriptureTranslationModel
from models.terminology import TerminologyDB
from inference import ScriptureTranslator
from evaluation import ScriptureEvaluator

# Client
client = ScriptureTranslationClient()
result = client.translate("In the beginning...", target_lang="spa_Latn")
print(result['primary'])

# Terminology
db = TerminologyDB()
db.add_term("salvation", "spa_Latn", "salvación")
print(db.lookup("salvation", "spa_Latn"))

# Model
model = ScriptureTranslationModel()
translator = ScriptureTranslator(model.get_model(), model.get_tokenizer(), db)

# Evaluate
evaluator = ScriptureEvaluator()

═══════════════════════════════════════════════════════════════════════════════
TROUBLESHOOTING
═══════════════════════════════════════════════════════════════════════════════

Problem: Out of Memory (OOM)
Solution: In config.py, reduce TRAINING_CONFIG["batch_size"] = 8

Problem: Slow Training
Solution: Ensure GPU: python -c "import torch; print(torch.cuda.is_available())"

Problem: Low Scores on Rare Language
Solution: Collect more data (target 1000+ verses), train longer (5-10 epochs)

Problem: Inconsistent Terminology
Solution: Build terminology database with native speakers, use enforce_consistency=True

Problem: API not responding
Solution: Check logs, ensure app.py is running, verify port 5000 is open
